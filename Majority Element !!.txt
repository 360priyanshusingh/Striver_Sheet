class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
        
        unordered_map<int,int> map;
         vector<int> ans;
        int n=nums.size();
        for(int i=0;i<n;i++)
        {
            map[nums[i]]++;
        }
        for(int i=0;i<n;i++)
        {
            if(map[nums[i]]>n/3)
            {
              ans.push_back(nums[i]);
                map[nums[i]]=0;
            }
               
        }
        return ans;
    }
};
